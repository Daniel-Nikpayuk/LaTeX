\begin{asydef}
// this comment prevents a compilation bug.

import three;

triple camera_angle = (-1, -1, 1);

currentprojection = perspective(camera = camera_angle);

void draw_axes(real scalar = 5)
{
	triple X3 = scalar*X;
	triple Y3 = scalar*Y;
	triple Z3 = scalar*Z;

	draw(O--X3, red + dotted, EndArrow3);
	draw(O--Y3, red + dotted, EndArrow3);
	draw(O--Z3, red + dotted, EndArrow3);

	label("$ x $", X3, red);
}

void draw_ground(real scalar = 5)
{
	triple DL3 = scalar*(-1, -1, 0);
	triple DR3 = scalar*( 1, -1, 0);
	triple UR3 = scalar*( 1,  1, 0);
	triple UL3 = scalar*(-1,  1, 0);

	draw(DL3--DR3--UR3--UL3--cycle, red + dotted);
}

\end{asydef}
